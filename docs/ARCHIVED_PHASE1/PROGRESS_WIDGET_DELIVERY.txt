USAGE EXAMPLES
================================================================================

Basic Usage:
───────────
from src.ui.progress_widget import DevelopmentProgressWidget

widget = DevelopmentProgressWidget()
widget.update_progress(5, "Design PyQt6 main window layout", "UI Design")
layout.addWidget(widget)


With Progress Manager:
──────────────────────
from src.ui.progress_manager import get_progress_manager
from src.ui.progress_widget import DevelopmentProgressWidget

manager = get_progress_manager()
widget = DevelopmentProgressWidget()

manager.signals.progress_updated.connect(widget.update_progress)
manager.signals.project_completed.connect(widget.set_success_state)

# Start a task
manager.start_task(5)

# Do work...

# Complete task
manager.complete_task(5)


Error Handling:
───────────────
widget.set_error_state("Failed to load model")
# Shows: "❌ Error: Failed to load model"
# Shows: "Please check the logs"
# Progress bar turns red


Success State:
──────────────
widget.set_success_state()
# Shows: "✓ Development Complete!"
# Shows: "All tasks completed successfully"
# Progress bar turns green
